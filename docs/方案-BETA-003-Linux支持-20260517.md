#### 2.3.2 `detectShells()` — 读 `/etc/shells`

```ts
async detectShells(): Promise<ShellInfo[]> {
  try {
    const content = await fs.promises.readFile('/etc/shells', 'utf8');
    const candidates = content
      .split('\n')
      .map((line) => line.trim())
      .filter((line) => line && !line.startsWith('#'));

    const results: ShellInfo[] = [];
    for (const path of candidates) {
      // 过滤伪 shell:/usr/sbin/nologin / /bin/false / /sbin/halt 等
      const base = basename(path);
      if (['nologin', 'false', 'halt', 'sync', 'shutdown'].includes(base)) continue;

      try {
        await fs.promises.access(path, fs.constants.X_OK);
      } catch {
        continue; // shell 不存在或不可执行,跳过
      }

      const id = base; // 'bash' / 'zsh' / 'fish'
      const displayName = SHELL_DISPLAY_NAMES[id] ?? base;
      results.push({ id, displayName, executablePath: path });
    }

    // 优先级:zsh > fish > bash > 其他(zsh 现代用户首选,fish 次之,bash 兜底)
    results.sort((a, b) => priorityOf(a.id) - priorityOf(b.id));
    return results;
  } catch (err) {
    logger.warn('linux-adapter', 'detectShells failed', err);
    // 兜底:返回 bash 假设它在 /bin/bash
    return [{ id: 'bash', displayName: 'Bash', executablePath: '/bin/bash' }];
  }
}

const SHELL_DISPLAY_NAMES: Record<string, string> = {
  bash: 'Bash',
  zsh: 'Zsh',
  fish: 'Fish',
  dash: 'Dash',
  ksh: 'KornShell',
  tcsh: 'TCSH',
  csh: 'C Shell',
};

function priorityOf(id: string): number {
  return { zsh: 0, fish: 1, bash: 2 }[id] ?? 10;
}
```

#### 2.3.3 `buildShellLaunchParams()` — 注入 OSC 1337 hook

每个 shell 启动方式不同,但 hook 注入心智一致:**让 shell 在执行用户 rc 文件之前 / 之后 source 一段 Marina 提供的 hook 脚本**(报告 cwd via OSC 1337)。

```ts
buildShellLaunchParams(
  shell: ShellInfo,
  hookFilePath: string,
  commandToRun?: { command: string; args: string[] },
): { args: string[]; env: Record<string, string> } {
  const env: Record<string, string> = {};
  let args: string[] = [];

  switch (shell.id) {
    case 'bash': {
      // bash 走 --rcfile 指向 hook,hook 内部最后 source 用户原 ~/.bashrc
      args = ['--rcfile', hookFilePath];
      if (commandToRun) {
        // bash --rcfile X -c "cmd args" 会先 source X 再跑 -c
        args.push('-c', shellQuote([commandToRun.command, ...commandToRun.args]));
      } else {
        args.push('-i'); // 交互模式
      }
      break;
    }
    case 'zsh': {
      // zsh 读 ZDOTDIR/.zshrc;hook 文件被放在临时目录,用 ZDOTDIR 指过去
      env['ZDOTDIR'] = dirname(hookFilePath);
      // hook 文件必须命名为 .zshrc,临时目录里其他没有
      // 由 src/shell-hooks/zsh-hook.zshrc 复制
      args = commandToRun
        ? ['-c', shellQuote([commandToRun.command, ...commandToRun.args])]
        : ['-i'];
      break;
    }
    case 'fish': {
      // fish 读 XDG_CONFIG_HOME/fish/config.fish,hook 走自定义 config 目录
      env['XDG_CONFIG_HOME'] = dirname(hookFilePath);
      args = commandToRun ? ['-c', shellJoin([commandToRun.command, ...commandToRun.args])] : ['-i'];
      break;
    }
    default: {
      // 未知 shell:无 hook,直接起交互式
      args = commandToRun ? ['-c', commandToRun.command + ' ' + commandToRun.args.join(' ')] : [];
    }
  }

  return { args, env };
}
```

**shell-hook 文件**(新增):

- `src/shell-hooks/bash-hook.bashrc` — 已有(Git Bash on Windows 用),Linux 直接复用,确认 PROMPT_COMMAND 注入兼容 POSIX bash

#### 2.3.4 `getProcessCwd()` — `readlink /proc/<pid>/cwd`

```ts
async getProcessCwd(pid: number): Promise<string | null> {
  if (!Number.isInteger(pid) || pid <= 0) return null;
  try {
    const target = await fs.promises.readlink(`/proc/${pid}/cwd`);
    return target;
  } catch (err) {
    // ENOENT(进程已退出)/ EACCES(权限不足,通常不发生因为是自己起的子进程)
    logger.debug('linux-adapter', `getProcessCwd(${pid}) failed`, err);
    return null;
  }
}
```

权限说明:对自己创建的子进程,readlink 自身 cwd 不需要 root;`/proc/<pid>/cwd` 在同一 UID 下可读。

#### 2.3.5 `setAutoStart()` / `isAutoStartEnabled()` — XDG autostart

```ts
private readonly autostartPath = join(
  process.env.XDG_CONFIG_HOME ?? join(homedir(), '.config'),
  'autostart',
  'marina.desktop',
);

async setAutoStart(enabled: boolean): Promise<void> {
  if (enabled) {
    const exePath = process.execPath;
    const content = [
      '[Desktop Entry]',
      'Type=Application',
      'Name=Marina',
      `Exec=${exePath}`,
      'Hidden=false',
      'NoDisplay=false',
      'X-GNOME-Autostart-enabled=true',
      '',
    ].join('\n');
    await fs.promises.mkdir(dirname(this.autostartPath), { recursive: true });
    await fs.promises.writeFile(this.autostartPath, content, { mode: 0o644 });
  } else {
    try {
      await fs.promises.unlink(this.autostartPath);
    } catch (err) {
      if ((err as NodeJS.ErrnoException).code !== 'ENOENT') throw err;
    }
  }
}

async isAutoStartEnabled(): Promise<boolean> {
  try {
    await fs.promises.access(this.autostartPath, fs.constants.F_OK);
    return true;
  } catch {
    return false;
  }
}
```

#### 3.3.3 主进程触发逻辑

`src/main/index.ts:252-260` 的 `window-all-closed` 重写为按 `lifecycleModel` 分支:

```ts
import { getPlatformAdapter } from './platform';
import { sessionManager } from './session-manager';
import { COMMAND_CHANNELS } from '../shared/protocol';

app.on('window-all-closed', () => {
  if (isQuitting) return;

  const adapter = getPlatformAdapter();
  switch (adapter.lifecycleModel) {
    case 'tray-resident':
      // Windows:不退出,沉托盘(已有行为)
      return;
    case 'dock-resident':
      // macOS:不退出,Electron darwin 默认就不退(此 if 实际不会触发,留作显式)
      return;
    case 'no-persistence': {
      // Linux:检查是否还有 alive session
      const aliveCount = sessionManager.list().filter((s) => s.state !== 'exited').length;
      if (aliveCount === 0) {
        app.quit();
        return;
      }
      // 有 alive session:弹 modal 由 renderer 决定
      // 但此时所有窗口已关闭...策略:在最后一个窗口的 close 事件里拦截,而不是 window-all-closed
      // 详见下方"为什么改在 close 事件"
      return;
    }
  }
});
```

**为什么改在 `BrowserWindow.close` 事件**(关键设计修正):

`window-all-closed` 触发时窗口已全部 destroy,renderer 已不在,**没法弹 modal**。正确的做法是在**最后一个窗口的 `close` 事件**里 `event.preventDefault()`,显示 modal 后再决定是否真关。

```ts
// src/main/window-manager.ts(假设此文件存在,否则在 index.ts 创建窗口时挂)
function attachLifecycleGuard(win: BrowserWindow) {
  win.on('close', (event) => {
    if (isQuitting) return; // 已经在退出流程
    const adapter = getPlatformAdapter();
    if (adapter.lifecycleModel !== 'no-persistence') return;

    // 检查这是不是"最后一个窗口"
    const otherOpenWindows = BrowserWindow.getAllWindows().filter(
      (w) => w !== win && !w.isDestroyed(),
    );
    if (otherOpenWindows.length > 0) return; // 不是最后一个,放过

    // 检查 alive session
    const aliveSessions = sessionManager.list().filter((s) => s.state !== 'exited');
    if (aliveSessions.length === 0) return; // 全 exited,静默退

    // 阻止关闭,触发 modal
    event.preventDefault();
    win.webContents.send(COMMAND_CHANNELS.UI_SHOW_LAST_SESSION_CONFIRM, {
      sessionCount: aliveSessions.length,
    });
  });
}
```

renderer 侧收到 IPC → dispatch `view/show-last-session-confirm` → `<App>` 渲染 modal → 用户点确认 → invoke `cmd:app:quit` → 主进程 `isQuitting = true; app.quit()`;用户点取消 → modal 关闭,什么都不做(窗口仍开)。

#### 3.3.4 Windows / macOS 复用同一 modal

- **Windows**:`src/main/tray.ts` 托盘菜单"完全退出"按钮当前用 native dialog,改成给 main window 发 IPC 触发同 modal。若所有窗口已沉托盘(无可见窗口),先 `mainWindow.show()` 再触发。

```ini
Exec=marina
```

字段说明:
- `Categories=System;TerminalEmulator;` — freedesktop 标准,告诉文件管理器"这是一个终端"
- `Exec=marina --working-directory=%f` — Nautilus 调用 `%f` 替换为文件夹绝对路径
- `Icon=marina` — 由打包脚本拷贝 PNG 到 `/usr/share/icons/hicolor/256x256/apps/marina.png`
- `StartupWMClass=Marina` — 任务栏图标分组用,与窗口 WM_CLASS 对齐

#### 4.2.3 `LinuxAdapter.registerFileManagerIntegration`

```ts
async registerFileManagerIntegration(appExePath: string): Promise<void> {
  // GNOME / GTK 系列:gsettings 设默认终端
  // 兼容老 GNOME (≤43) 的 org.gnome.desktop.default-applications.terminal
  // 新 GNOME (44+) 走 xdg-terminal-exec spec,gsettings 不再生效,但 .desktop
  // 的 Categories=TerminalEmulator 在 XDG_TERMINAL_EXEC=marina 时被读
  try {
    await execFileAsync('gsettings', [
      'set',
      'org.gnome.desktop.default-applications.terminal',
      'exec',
      'marina',
    ]);
  } catch (err) {
    logger.warn('linux-adapter', 'gsettings set failed (non-GNOME?)', err);
  }

  // Debian/Ubuntu 的 alternatives:让所有走 x-terminal-emulator 的 GTK 程序找到 marina
  try {
    await execFileAsync('update-alternatives', [
      '--set',
      'x-terminal-emulator',
      '/usr/bin/marina',
    ]);
  } catch (err) {
    // RHEL/CentOS/Fedora 用 alternatives(不带 update-)
    try {
      await execFileAsync('alternatives', ['--set', 'x-terminal-emulator', '/usr/bin/marina']);
    } catch (err2) {
      logger.warn('linux-adapter', 'update-alternatives / alternatives both failed', err2);
    }
  }
}

async unregisterFileManagerIntegration(): Promise<void> {
  // gsettings 回退到 gnome-terminal 默认
  try {
    await execFileAsync('gsettings', [
      'reset',
      'org.gnome.desktop.default-applications.terminal',
      'exec',
    ]);
  } catch (err) {
    logger.warn('linux-adapter', 'gsettings reset failed', err);
  }
  // alternatives 切回 auto 模式
  try {
    await execFileAsync('update-alternatives', ['--auto', 'x-terminal-emulator']);
  } catch (err) {
    try {
      await execFileAsync('alternatives', ['--auto', 'x-terminal-emulator']);
    } catch {
      /* 同上 */
    }
  }
}
```

#### 5.2.2 postinst / postrm 脚本

`build/linux/deb-postinst.sh`:

```sh
#!/bin/sh
set -e
# 注册 x-terminal-emulator alternative,优先级 50(系统默认 gnome-terminal 是 40)
update-alternatives --install /usr/bin/x-terminal-emulator x-terminal-emulator /usr/bin/marina 50
exit 0
```

`build/linux/deb-postrm.sh`:

```sh
#!/bin/sh
set -e
if [ "$1" = "remove" ] || [ "$1" = "purge" ]; then
  update-alternatives --remove x-terminal-emulator /usr/bin/marina
fi
exit 0
```

`build/linux/rpm-postinst.sh` 与 `rpm-postrm.sh`:把 `update-alternatives` 改成 `alternatives`,语义一致。

### 5.4 SELinux

CentOS / RHEL / Fedora 默认 enforcing。Marina 的 PTY spawn + Chromium sandbox 在 SELinux 下可能被拦。

**验证步骤**:在 Fedora VM:

```sh

```

如果有 marina 相关 deny,生成 policy module 加到 `.rpm` spec。如果干净,**什么都不做**。**待 003d 实际验证后才知道**。
